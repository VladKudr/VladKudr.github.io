Аутентификация по паре логин/пароль согласно ПИР 31

Что планируем сделать

В задаче аутентификации по паре логин/пароль самой трудной частью является расчет свертки пароля, поэтому мы уделим ей наиболее пристальное внимание, но пройдем по также по всему процессу получения токена сессии. Предлагаю также вам открыть спецификацию форматов и порядка обмена данными от Сбербанка.

Вызов метода preLogin

Вы выполняете первым шагом вызов preLogin со структурой аналогичной ниже:

   <soapenv:Body>

Расчет свертки SRP

Не будем повторять инструкцию из спецификации, разберем как выполнить расчет программным образом

Подсчет расчета Java
public class SRPService {

    public List<byte[]> getKandAWNew(String login, String password, byte[] Salt, byte[] B_arr) {
        List<byte[]> result = new ArrayList<byte[]>();
        BigInteger N, g;

        //Определение констант
        N = new BigInteger(
                "0115B8B692E0E045692CF280B436735C77A5A9E8A9E7ED56C965F87DB5B2A2ECE3",
                16);
        g = new BigInteger("2", 16);

        int gLength = g.toByteArray().length;
        int NLength = N.toByteArray().length * 8;
        BigInteger B = new BigInteger(1, B_arr);

        //Фиксируем значение для теста, чтобы получать всегда один и тот же результат
        BigInteger a = new BigInteger("D0F6A2EC715AD49030FB4BECAB68079B614B1B8631B3DC17CEE141BCE0DA8D5D", 16);

        BigInteger A = g.modPow(a, N);


        //Инициализация алгоритма хеширования
        MessageDigest digest;
        try {
            digest = MessageDigest.getInstance("SHA-1");
        } catch (NoSuchAlgorithmException e) {
            throw new RuntimeException(e.getMessage(), e);
        }


        digest.update(N.toByteArray());

        for (int i = (NLength >>> 3) - gLength; i > 0; i--)
            digest.update((byte) 0);
        digest.update(g.toByteArray());

        byte[] kBytes = digest.digest();
        BigInteger k = new BigInteger(1, kBytes);

        BigInteger vs = new BigInteger(1, Salt);

        digest.reset();


        byte[] vUserPasswordHash = digest.digest(trimBytePaddingLeft((login + ":" + (password == null ? "" : password.trim())).getBytes(Charset.forName("UTF-8"))));
        digest.reset();

        digest.update(trimBytePaddingLeft(vs.toByteArray()));
        digest.update(trimBytePaddingLeft(vUserPasswordHash));
        BigInteger vx = new BigInteger(1, trimBytePaddingLeft(digest.digest()));

        if (vx.compareTo(N) > 0) {
            vx = vx.mod(N);
        }


        digest.reset();

        for (int i = (NLength >>> 3) - A.toByteArray().length; i > 0; i--)
            digest.update((byte) 0);

        digest.update(A.toByteArray());


        for (int i = (NLength >>> 3) - B.toByteArray().length; i > 0; i--)
            digest.update((byte) 0);

        digest.update(B.toByteArray());
        byte[] uArr = digest.digest();

        BigInteger u = new BigInteger(1, uArr);

        if (u.compareTo(N) > 0) {
            u = u.mod(N);
        }


        BigInteger pGN = g.modPow(vx, N);


        BigInteger kMult = k.multiply(pGN);
        BigInteger base = B.subtract(kMult);


        if (base.signum() < 0) {
            base = base.add(N.multiply(k));
        }


        BigInteger exp = a.add(u.multiply(vx));


        BigInteger Sb = base.modPow(exp, N);


        byte[] SIntByte = bigIntToBytes(Sb);


        byte[] vk = digest.digest(SIntByte);
        BigInteger K = new BigInteger(vk);


        result.add(trimBytePaddingLeft(K.toByteArray()));
        result.add(trimBytePaddingLeft(A.toByteArray()));

        return result;
    }
}
